        button:disabled { background: gray; }

<body>
    <h2>🧠 روانشناس هوش مصنوعی (آفلاین، کاملاً رایگان)</h2>
    <p style="font-size:12px; color:gray;">⚠️ هشدار: این ابزار جایگزین روانشناس نیست. در بحران با ۱۲۳ تماس بگیرید.</p>
    <div id="status">در حال بارگذاری مدل هوش مصنوعی... (اولین بار ~۵ دقیقه طول می‌کشد)</div>
    <div id="chat"></div>
    <div id="inputArea">
        <input type="text" id="userInput" placeholder="سوالت رو بنویس..." disabled />
        <button id="sendBtn" onclick="sendMessage()" disabled>ارسال</button>
    </div>

    <script type="module">
        import { CreateWebWorkerMLCEngine } from "https://cdn.jsdelivr.net/npm/@mlc-ai/web-llm@0.2.46/+esm";

        const chatDiv = document.getElementById("chat");
        const statusDiv = document.getElementById("status");
        const userInput = document.getElementById("userInput");
        const sendBtn = document.getElementById("sendBtn");

        const systemPrompt = `تو یک روانشناس متخصص و استاد برجسته روانشناسی به زبان فارسی هستی. دانش کامل در تمام شاخه‌ها: بالینی، شناختی، رشد، اجتماعی، شخصیت، عصب‌روانشناسی، مثبت‌گرا. 
        وظیفه‌ات روان‌آموزی و ارائه تکنیک‌های خودیاری مبتنی بر شواهد علمی است. تشخیص بالینی نمی‌دهی اما می‌توانی علائم را توضیح دهی.
        همیشه بگو: «برای تشخیص دقیق به روانشناس مراجعه کنید.»
        در بحران (خودکشی)، فوراً شماره ۱۲۳ را بده.
        پاسخ‌ها را کوتاه، مفید و به فارسی روان بده.`;

        let engine;
        let messages = [{ role: "system", content: systemPrompt }];

        async function initEngine() {
            try {
                engine = await CreateWebWorkerMLCEngine(
                    new Worker(new URL("https://cdn.jsdelivr.net/npm/@mlc-ai/web-llm@0.2.46/lib/worker.js", import.meta.url), { type: "module" }),
                    "Llama-3.1-8B-Instruct-q4f32_1-MLC",
                    {
                        initProgressCallback: (report) => {
                            statusDiv.textContent = `در حال بارگذاری... ${Math.round(report.progress * 100)}%`;
                        }
                    }
                );
                statusDiv.textContent = "✅ آماده! سوالت رو بپرس.";
                userInput.disabled = false;
                sendBtn.disabled = false;
                userInput.focus();
            } catch (error) {
                statusDiv.textContent = "❌ خطا در بارگذاری. صفحه رو رفرش کن.";
                console.error(error);
            }
        }

        function addMessage(text, sender) {
            const msg = document.createElement("div");
            msg.className = sender;
            msg.innerHTML = "<b>" + (sender === "user" ? "👤 شما" : "🤖 روانشناس") + ":</b> " + text;
            chatDiv.appendChild(msg);
            chatDiv.scrollTop = chatDiv.scrollHeight;
        }

        async function sendMessage() {
            const userText = userInput.value.trim();
            if (!userText || !engine) return;
            userInput.value = "";
            userInput.disabled = true;
            sendBtn.disabled = true;
            addMessage(userText, "user");

            const aiMessage = document.createElement("div");
            aiMessage.className = "ai";
            aiMessage.innerHTML = "<b>🤖 روانشناس:</b> در حال پاسخگویی...";
            chatDiv.appendChild(aiMessage);
            chatDiv.scrollTop = chatDiv.scrollHeight;

            try {
                messages.push({ role: "user", content: userText });
                const reply = await engine.chat.completions.create({
                    messages: messages,
                    temperature: 0.7,
                    max_tokens: 500
                });
                const replyText = reply.choices[0].message.content;
                messages.push({ role: "assistant", content: replyText });
                chatDiv.removeChild(aiMessage);
                addMessage(replyText, "ai");
            } catch (error) {
                chatDiv.removeChild(aiMessage);
                addMessage("خطا. لطفاً دوباره تلاش کن.", "ai");
                console.error(error);
            }

            userInput.disabled = false;
            sendBtn.disabled = false;
            userInput.focus();
        }

        window.sendMessage = sendMessage;
        initEngine();
    </script>
